  方案对比

  方案 A: 单 HTML 文件 + Canvas（推荐）
  - 所有代码写在一个 index.html 里
  - 使用 HTML5 Canvas 渲染，像素风格完美还原
  - 双击即可运行，无需服务器
  - 优点：最简单，一个文件搞定，适合 GitHub Pages 直接部署
  - 缺点：代码都在一个文件里，维护性一般

  方案 B: 多文件项目 + Canvas
  - 分离 HTML / CSS / JS 文件
  - 使用 Canvas 渲染
  - 优点：代码组织清晰
  - 缺点：需要本地服务器运行（或配置 GitHub Pages），文件多一些

  方案 C: 单 HTML 文件 + DOM 元素
  - 用 div 做游戏对象，CSS 做样式
  - 优点：CSS 动画效果好
  - 缺点：性能差，像素风格还原度低

  我推荐方案 A，因为经典坦克大战用 Canvas 最合适，单文件方便分享和部署。

● 游戏设计

  核心玩法

  - 地图: 13x13 的网格系统，经典坦克大战布局
  - 玩家坦克: 方向键移动，空格射击
  - 敌方坦克: AI 控制，从地图顶部生成，自动移动和射击
  - 基地: 地图底部中央的鹰标，被摧毁则游戏结束
  - 目标: 消灭所有敌方坦克过关

  地形元素

  ┌──────┬──────────────────────────────────┐
  │ 元素 │               说明               │
  ├──────┼──────────────────────────────────┤
  │ 砖墙 │ 可被子弹摧毁                     │
  ├──────┼──────────────────────────────────┤
  │ 钢墙 │ 不可摧毁                         │
  ├──────┼──────────────────────────────────┤
  │ 河流 │ 坦克不可通过，子弹可飞越         │
  ├──────┼──────────────────────────────────┤
  │ 草地 │ 坦克可通行，覆盖坦克（隐藏效果） │
  ├──────┼──────────────────────────────────┤
  │ 冰面 │ 坦克可通行，移动时会滑行         │
  └──────┴──────────────────────────────────┘

  道具系统

  - 星星: 升级坦克（射速/速度提升）
  - 炸弹: 消灭当前所有敌方坦克
  - 护盾: 临时无敌
  - 铲子: 临时加固基地周围为钢墙

  技术实现

  - 渲染: Canvas 2D，16x16 像素块绘制
  - 帧率: 60fps 游戏循环
  - 碰撞检测: 基于网格的碰撞系统
  - 音效: 无（保持单文件简洁）
  - 存档: 无

  关卡

  - 预设 5 个关卡，不同地图布局
  - 每关 20 个敌方坦克
  - 通关后自动进入下一关

   378 // ===== 输入处理 =====
   379 function handleKey(key) {
   380   if (key === 'p' || key === 'P') {
   381     if (gameState === 'playing') gameState = 'paused';
   382     else if (gameState === 'paused') gameState = 'playing';
   383   }
   384   if (key === 'r' || key === 'R') {
   385     if (gameState === 'gameover' || gameState === 'win') {
   386       score = 0;
   387       lives = 3;
   388       level = 1;
   389       loadLevel(level);
   390       gameState = 'playing';
   391     }
   392   }
   393   if (key === 'Enter' && gameState === 'menu') {
   394     startGame();
   395   }
   396 }

   421 function updatePlayer() {
   422   if (!player.alive) return;
   423
   424   let moved = false;
   425   let newDir = player.dir;
   426
   427   if (keys['ArrowUp'] || keys['w'] || keys['W']) { newDir = UP; moved = tr
       ue; }
   428   else if (keys['ArrowDown'] || keys['s'] || keys['S']) { newDir = DOWN; m
       oved = true; }
   429   else if (keys['ArrowLeft'] || keys['a'] || keys['A']) { newDir = LEFT; m
       oved = true; }
   430   else if (keys['ArrowRight'] || keys['d'] || keys['D']) { newDir = RIGHT;
        moved = true; }
   431
   432   player.dir = newDir;
   433
   434   if (moved) {
   435     const nx = player.x + DX[newDir] * player.speed;
   436     const ny = player.y + DY[newDir] * player.speed;
   437     if (canMove(nx, ny, player.size, player)) {
   438       player.x = nx;
   439       player.y = ny;
   440     }
   441     player.animFrame++;
   442   }
   443
   444   if (player.shootCooldown > 0) player.shootCooldown--;
   445   if (keys[' '] && player.shootCooldown <= 0) {
   446     shoot(player);
   447     player.shootCooldown = 20;
   448   }
   449
   450   if (shieldTimer > 0) shieldTimer--;
   451   if (shovelTimer > 0) {
   452     shovelTimer--;
   453     if (shovelTimer === 0) {
   454       // 恢复基地砖墙
   455       protectBase();
   456       const baseY = 12, baseX = 6;
   457       const protection = [
   458         [baseY-1, baseX-1], [baseY-1, baseX], [baseY-1, baseX+1],
   459         [baseY, baseX-1], [baseY, baseX+1],
   460       ];
   461       for (const [r, c] of protection) {
   462         if (map[r][c] === STEEL) map[r][c] = BRICK;
   463       }
   464     }
   465   }

   526         continue;
   527       }
   528       if (tile === WATER) {
   529         // 子弹飞越水面
   530       }
   531       if (tile === BASE) {
   532         map[row][col] = BASE_DEAD;
   533         bullets.splice(i, 1);

   607 function canMove(x, y, size, self) {
   608   // 边界检查
   609   if (x < 0 || x + size > W || y < 0 || y + size > H) return false;
   610
   611   // 地图碰撞
   612   const left = Math.floor(x / TILE);
   613   const top = Math.floor(y / TILE);
   614   const right = Math.floor((x + size - 1) / TILE);
   615   const bottom = Math.floor((y + size - 1) / TILE);
   616
   617   for (let r = top; r <= bottom; r++) {
   618     for (let c = left; c <= right; c++) {
   619       if (r < 0 || r >= ROWS || c < 0 || c >= COLS) continue;
   620       const tile = map[r][c];
   621       if (tile === BRICK || tile === STEEL || tile === WATER || tile === B
       ASE) {
   622         return false;
   623       }
   624     }
   625   }
   626
   627   // 坦克碰撞
   628   const allTanks = [player, ...enemies].filter(t => t && t.alive && t !==
       self);
   629   for (const t of allTanks) {
   630     if (rectsOverlap(x, y, size, size, t.x, t.y, t.size, t.size)) {
   631       return false;
   632     }
   633   }
   634
   635   return true;
   636 }

   638 function rectsOverlap(x1, y1, w1, h1, x2, y2, w2, h2) {
   639   return x1 < x2 + w2 && x1 + w1 > x2 && y1 < y2 + h2 && y1 + h1 > y2;
   640 }

   642 function checkCollisions() {
   643   // 子弹 vs 坦克
   644   for (let i = bullets.length - 1; i >= 0; i--) {
   645     const b = bullets[i];
   646
   647     if (b.owner === 'player') {
   648       // 玩家子弹 vs 敌人
   649       for (const e of enemies) {
   650         if (!e.alive) continue;
   651         if (b.x >= e.x && b.x <= e.x + e.size && b.y >= e.y && b.y <= e.y
       + e.size) {
   652           e.hp--;
   653           bullets.splice(i, 1);
   654           if (e.hp <= 0) {
   655             e.alive = false;
   656             enemiesRemaining--;
   657             score += 100;
   658             addExplosion(e.x + TILE/2, e.y + TILE/2, 'big');
   659
   660             // 随机掉落道具
   661             if (Math.random() < 0.2) {
   662               spawnPowerup();
   663             }
   664           } else {
   665             addExplosion(b.x, b.y, 'tiny');
   666           }
   667           updateUI();
   668           break;
   669         }
   670       }
   671     } else {
   672       // 敌人子弹 vs 玩家
   673       if (player.alive && shieldTimer <= 0) {
   674         if (b.x >= player.x && b.x <= player.x + player.size &&
   675             b.y >= player.y && b.y <= player.y + player.size) {
   676           bullets.splice(i, 1);
   677           playerHit();
   678           break;
   679         }
   680       }
   681     }
   682
   683     // 子弹 vs 子弹
   684     for (let j = bullets.length - 1; j >= 0; j--) {
   685       if (i === j) continue;
   686       const b2 = bullets[j];
   687       if (Math.abs(b.x - b2.x) < 6 && Math.abs(b.y - b2.y) < 6) {
   688         if (i > j) { bullets.splice(i, 1); bullets.splice(j, 1); }
   689         else { bullets.splice(j, 1); bullets.splice(i, 1); }
   690         i = Math.min(i, j) - 1;
   691         break;
   692       }
   693     }
   694   }
   695
   696   // 玩家 vs 道具
   697   for (let i = powerups.length - 1; i >= 0; i--) {
   698     const p = powerups[i];
   699     if (player.alive && rectsOverlap(player.x, player.y, player.size, play
       er.size,
   700         p.x, p.y, TILE, TILE)) {
   701       applyPowerup(p.type);
   702       powerups.splice(i, 1);
   703     }
   704   }
   705 }

   709   addExplosion(player.x + TILE/2, player.y + TILE/2, 'big');

   714     // 重生
   715     player.x = 4 * TILE;
   716     player.y = 12 * TILE;
   717     player.dir = UP;
   718     shieldTimer = 120;
   719   }
   720   updateUI();
   721 }

   744         }
   745       }
   746       break;
   747     case 'shield':
   748       shieldTimer = 300;
   749       break;
   750     case 'shovel':
   751       shovelTimer = 600;
   752       const baseY = 12, baseX = 6;
   753       const protection = [
   754         [baseY-1, baseX-1], [baseY-1, baseX], [baseY-1, baseX+1],
   755         [baseY, baseX-1], [baseY, baseX+1],
   756       ];
   757       for (const [r, c] of protection) {
   758         if (map[r][c] === BRICK || map[r][c] === EMPTY) {
   759           map[r][c] = STEEL;
   760         }

   762       break;
